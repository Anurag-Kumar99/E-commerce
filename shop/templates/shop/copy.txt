<script>
// Cart initialization
let cart = JSON.parse(localStorage.getItem('cart')) || {};

// Update cart display on page load
updateCartDisplay();

// Handle all cart button clicks
$(document).on('click', '.cart, .plus, .minus', function(e) {
    e.preventDefault();
    const button = $(this);
    const action = button.hasClass('cart') ? 'add' : 
                  button.hasClass('plus') ? 'increase' : 'decrease';
    
    const productId = button.attr('id').replace('pr', '').replace('plus', '').replace('minus', '');
    const productKey = `pr${productId}`;
    
    // Get product details
    const productName = $(`#namepr${productId}`).text();
    const productPrice = parseInt($(`#pricepr${productId}`).text().replace('Price: Rs.', '').trim());
    
    // Update cart based on action
    if (action === 'add') {
        cart[productKey] = cart[productKey] || [0, productName, productPrice];
        cart[productKey][0] += 1;
    } 
    else if (action === 'increase') {
        if (cart[productKey]) cart[productKey][0] += 1;
    } 
    else if (action === 'decrease') {
        if (cart[productKey]) {
            cart[productKey][0] -= 1;
            if (cart[productKey][0] <= 0) {
                delete cart[productKey];
            }
        }
    }
    
    // Save and update
    localStorage.setItem('cart', JSON.stringify(cart));
    updateCartDisplay();
});

function updateCartDisplay() {
    let totalItems = 0;
    
    // Update all product buttons
    $('.divpr').each(function() {
        const div = $(this);
        const productId = div.attr('id').replace('divpr', '');
        const productKey = `pr${productId}`;
        
        if (cart[productKey] && cart[productKey][0] > 0) {
            div.html(`
                <button id="minus${productId}" class="btn btn-primary minus">-</button>
                <span id="val${productId}">${cart[productKey][0]}</span>
                <button id="plus${productId}" class="btn btn-primary plus">+</button>
            `);
            totalItems += cart[productKey][0];
        } else {
            div.html(`<button id="pr${productId}" class="btn btn-primary cart">Add To Cart</button>`);
        }
    });
    
    // Update cart count
    $('#cart').text(totalItems);
    
    // Update popover
    updatePopover();
}

function updatePopover() {
    let popoverContent = '<h5>Your Cart</h5><div class="mx-2 my-2">';
    let total = 0;

    Object.keys(cart).forEach((key, index) => {
        // ðŸ›¡ Safe check
        if (cart[key] && Array.isArray(cart[key]) && cart[key].length >= 3 && cart[key][0] >= 0) {
            popoverContent += `
                <b>${index + 1}.</b> ${cart[key][1].slice(0, 19)}... 
                Qty: ${cart[key][0]} 
                (Rs.${cart[key][0] * cart[key][2]})<br>
            `;
            total += cart[key][0] * cart[key][2];
        }
    });

    popoverContent += `</div><b>Total: Rs.${total}</b><br>
        <a href="/shop/checkout" class="btn btn-primary btn-sm mt-2">Checkout</a>
        <button class="btn btn-danger btn-sm mt-2" onclick="clearCart()">Clear Cart</button>`;

    $('#popcart').attr('data-content', popoverContent);
}

function clearCart() {
    cart = {};
    localStorage.removeItem('cart');
    updateCartDisplay();
    $('#popcart').popover('hide');
}

// Initialize popover
$(document).ready(function() {
    $('#popcart').popover({
        html: true,
        trigger: 'click'
    });
});
</script>
